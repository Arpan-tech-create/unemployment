            list-style-type: none;

        }

        /* Styling for the burglar menu */
        .navbar .burglar-menu {
            display: none; /* Hide the burglar menu by default */
        }

        /* Styling for the expanded menu */
        .navbar ul.expanded {
            display: block;
        }

        /* Media query to display the horizontal menu on smaller screens */
        @media (max-width: 768px) {
            .navbar .burglar-menu {
                display: block; /* Show the burglar menu on smaller screens */
            }

            .navbar .menu {
                display: none; /* Hide the horizontal menu on smaller screens */
            }

            .navbar ul.expanded.vertical {
                display: block; /* Show the expanded vertical menu on smaller screens */
            }
        }

        .content {
            margin-top: 60px;
            padding: 20px;
        }
    </style>
    <script>
        // JavaScript to handle the burglar menu click event
        document.addEventListener("DOMContentLoaded", function() {
            var burglarMenu = document.querySelector(".burglar-menu");
            var menu = document.querySelector(".navbar ul");

            burglarMenu.addEventListener("click", function() {
                menu.classList.toggle("expanded");
            });
        });
    </script>
